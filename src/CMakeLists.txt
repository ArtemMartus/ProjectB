function (add_source name)
    cmake_policy (SET CMP0076 NEW)
    target_sources (${LIB_NAME} PRIVATE ${name}.cpp)
    target_sources (${LIB_NAME} PUBLIC  ${INCLUDE_DIR}/${name}.h)
endfunction ()
